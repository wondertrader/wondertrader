#1. 确定CMake的最低版本需求
CMAKE_MINIMUM_REQUIRED(VERSION 3.0.0)

#2. 确定工程名
PROJECT(WtDataStorageAD LANGUAGES CXX)
SET(CMAKE_CXX_STANDARD 11)

SET(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/../build_${PLATFORM}/${CMAKE_BUILD_TYPE}/bin)

#7. 添加源码
file(GLOB SRCS *.cpp)
file(GLOB HDRS *.h)

INCLUDE_DIRECTORIES(${INCS})
LINK_DIRECTORIES(${LNKS})
ADD_LIBRARY(WtDataStorageAD SHARED ${SRCS} ${HDRS})

SET(LIBS
	WTSUtils
)

IF (MSVC)
ELSE (GNUCC)
	LIST(APPEND LIBS
		dl
		pthread
		boost_filesystem
		)
	if(WIN32)
		LIST(APPEND LIBS iconv)
	ENDIF()
ENDIF()
TARGET_LINK_LIBRARIES(WtDataStorageAD ${LIBS})

IF (MSVC)
ELSE (GNUCC)
	SET_TARGET_PROPERTIES(WtDataStorageAD PROPERTIES
		CXX_VISIBILITY_PRESET hidden
		VISIBILITY_INLINES_HIDDEN 1
        LINK_FLAGS_RELEASE -s)
ENDIF ()
